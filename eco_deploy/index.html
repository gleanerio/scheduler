    2. cd dagster/implnets/deployment
    3. create a copy of envFile.env and edit env variables
    4. PROJECT=eco
    5. GLEANERIO_MINIO_ADDRESS ++
    6. GLEANERIO_GRAPH_URL, GLEANERIO_GRAPH_NAMESPACE
